# AI/voice/vision runtime dependencies for the 7.3 demo (mode: voice + LLM +
# advisory YOLO). These live in the dedicated virtualenv /opt/ai-venv and are
# bridged into the ROS nodes (which run under /usr/bin/python3) via PYTHONPATH.
# Install with: bash scripts/setup_ai_venv.sh
#
# HARD CONSTRAINT: numpy MUST stay < 2. ROS 2 Jazzy / cv_bridge C-extensions are
# compiled against numpy 1.x; numpy 2.x makes them fail to import. The setup
# script re-pins numpy<2 as the very last step in case a wheel pulled numpy 2.
#
# The versions in comments are the exact set verified working on
# Ubuntu 24.04 (WSL2) + ROS 2 Jazzy + Python 3.12 as of this commit.

numpy<2                     # tested: 1.26.4  (do NOT bump to >=2)

# --- LLM (LangChain + HuggingFace Inference Endpoint) ---
langchain-core              # tested: 1.3.2
langchain-huggingface       # tested: 1.2.2
huggingface-hub             # tested: 1.12.0  (provides `huggingface-cli`)

# --- ASR (Whisper via transformers) + audio capture ---
transformers                # tested: 5.6.2
torch                       # tested: 2.11.0+cpu (CPU wheels: see setup script)
sounddevice                 # tested: 0.5.5

# --- Perception (advisory YOLO) ---
ultralytics                 # tested: 8.4.41
opencv-python<4.12          # tested 4.13.0.92 worked, but <4.12 avoids pulling
                            # numpy>=2 as a transitive dependency (keeps numpy<2)
